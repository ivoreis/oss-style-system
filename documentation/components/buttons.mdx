# Examples

```jsx
import { Button } from 'oss-style-system';
import { ThemeProvider } from 'styled-components';

const theme = {};

const Component = () => (
  <ThemeProvider theme={theme}>
    <Button>I'm a button</Button>
  </ThemeProvider>
);
```
